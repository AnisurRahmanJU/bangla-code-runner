```py

// Factorial in Loop
ধরি নাম্বার = ৫;
ধরি ফ্যাক্টরিয়াল = ১;
লুপ (ধরি ক = ১; ক <= নাম্বার; ক++)
{
  ফ্যাক্টরিয়াল = ফ্যাক্টরিয়াল * ক;
}
দেখাও("৫ এর ফ্যাক্টরিয়াল: " + ফ্যাক্টরিয়াল);
```

```py
// Recursive Factorial
ফাংশন ফ্যাক্টরিয়াল(নাম্বার) 
{
  যদি (নাম্বার == ১)
  {
    ফেরত ১;
  }
  নাহলে
  { 
    ফেরত নাম্বার * ফ্যাক্টরিয়াল (নাম্বার-১);
  }
}
  
ধরি ফলাফল = ফ্যাক্টরিয়াল(৫);
দেখাও("৫ এর ফ্যাক্টরিয়াল: " + ফলাফল);
```
