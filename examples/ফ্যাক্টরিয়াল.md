```py

// Factorial in Loop
ধরি ফ্যাক্টরিয়াল = ৫;
ধরি ফলাফল = ১;
লুপ (ধরি ক = ১; ক <= ফ্যাক্টরিয়াল; ক++)
{
  ফলাফল = ফলাফল * ক;
}
দেখাও("৫ এর ফ্যাক্টরিয়াল: " + ফলাফল);
```

```py
// Recursive Factorial
ফাংশন ফ্যাক্টরিয়াল(নাম্বার) 
{
  যদি (নাম্বার == ১)
  {
    ফেরত ১;
  }
  নাহলে
  { 
    ফেরত নাম্বার * ফ্যাক্টরিয়াল (নাম্বার-১);
  }
}
  
ধরি ফলাফল = ফ্যাক্টরিয়াল(৫);
দেখাও("৫ এর ফ্যাক্টরিয়াল: " + ফলাফল);
```
