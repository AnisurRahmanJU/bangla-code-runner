### Without Newline
```py
ধরি ক;
ধরি খ;
ধরি গ = ইনপুট("ইংরেজিতে একটি নাম্বার নাও: ");

লুপ(ক = ১; ক <= গ; ক++)
{
   ধরি নিউলাইন = " ";
    লুপ(খ = ১; খ <= ক; খ++)
    {
         নিউলাইন =  নিউলাইন + "  " + খ;
    }
    দেখাও(নিউলাইন); 
}

```
### With Newline
```py
ধরি ক;
ধরি খ;
ধরি গ = ইনপুট("ইংরেজিতে একটি নাম্বার নাও: ");

লুপ(ক = ১; ক <= গ; ক++)
{
   ধরি নিউলাইন = " ";
    লুপ(খ = ১; খ <= ক; খ++)
    {
        নিউলাইন = নিউলাইন + "      " + খ;
    }
    দেখাও(নিউলাইন);
    // "\ন" হলো নিউলাইন
    দেখাও("\ন"); 
}
```
### Five Box Pattern
```py
ধরি ক;
ধরি খ;
ধরি গ = ৭;

লুপ (ক = ১ ; ক <= গ; ক++) {
ধরি নিউলাইন = " "; 
    লুপ (খ = ১; খ <= গ; খ++) {
        যদি (ক==৩ || খ==৩ || ক==গ-২ || খ==গ-২ || ক == খ || ((ক==১ || ক==২ || ক==৩ || ক==গ || ক==গ-১) && (খ==১ || খ==২ || খ==৩ || খ==গ || খ==গ-১)))
      {
            নিউলাইন += "*    ";
       } 
        নাহলে 
        {
            নিউলাইন += "      ";
        }
    }
    দেখাও(নিউলাইন);
}
```
