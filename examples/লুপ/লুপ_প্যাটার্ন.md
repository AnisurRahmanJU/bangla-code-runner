### Without Newline
```py
ধরি ক;
ধরি খ;
ধরি গ = ইনপুট("ইংরেজিতে একটি নাম্বার নাও: ");

লুপ(ক = ১; ক <= গ; ক++)
{
   ধরি লাইন = " ";
    লুপ(খ = ১; খ <= ক; খ++)
    {
         লাইন =  লাইন + "  " + খ;
    }
    দেখাও(লাইন); 
}

```
### With Newline
```py
ধরি ক;
ধরি খ;
ধরি গ = ইনপুট("ইংরেজিতে একটি নাম্বার নাও: ");

লুপ(ক = ১; ক <= গ; ক++)
{
   ধরি লাইন = " ";
    লুপ(খ = ১; খ <= ক; খ++)
    {
        লাইন = লাইন + "      " + খ;
    }
    দেখাও(লাইন);
    // "\ন" হলো নিউলাইন
    দেখাও("\ন"); 
}
```
### Five Box Pattern
```py
ধরি ক;
ধরি খ;
ধরি গ = ইনপুট(" ইংরেজিতে একটি সংখ্যা লিখুন: ");  // ৫ এর উপরে বিজোড় নাম্বার

লুপ (ক = ১ ; ক <= গ; ক++) {
ধরি লাইন = ""; 
    লুপ (খ = ১; খ <= গ; খ++) {
        যদি (ক==৩ || খ==৩ || ক==গ-২ || খ==গ-২ || ((ক==১ || ক==২ || ক==৩ || ক==গ || ক==গ-১) && (খ==১ || খ==২ || খ==৩ || খ==গ || খ==গ-১)))
      {
            লাইন += "*    ";
       } 
        নাহলে 
        {
            লাইন += "      ";
        }
    }
    দেখাও(লাইন);
}
```
