```py

ধরি সংখ্যা = ৭;
ধরি ক = নাল 

// 'এবং' (AND) কন্ডিশন
যদি (সংখ্যা > ৫ এবং ক == নাল) {
  দেখাও("ঠিক আছে");
}
নাহলে {
    দেখাও("ঠিক নাই");
}

// 'অথবা' (OR) কন্ডিশন
যদি (সংখ্যা < ৫ অথবা ক == নাল) {
  দেখাও("ঠিক আছে");
}
নাহলে {
    দেখাও("ঠিক নাই");
}

// 'নাল' (NULL) কন্ডিশন
যদি (ক == নাল) {
  দেখাও("ঠিক আছে");
}
নাহলে {
    দেখাও("ঠিক নাই");
}
```
