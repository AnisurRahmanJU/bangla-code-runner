```py

ধরি সংখ্যা = ৭;
ধরি ক = নাল 

// 'এবং' (AND) কন্ডিশন
যদি (সংখ্যা > ৫ এবং ক == নাল) {
  দেখাও("সংখ্যাটি ৫ এর বেশি, ক হলো নাকচ");
}

// 'অথবা' (OR) কন্ডিশন
যদি (সংখ্যা < ৫ অথবা ক == নাল) {
  দেখাও("সংখ্যাটি ৫ এর কম, ক হলো নাকচ");
}

// 'নাল' (NULL) কন্ডিশন
যদি (ক == নাল) {
  দেখাও("ক হলো নাকচ");
}
```
