```py
ধরি রং_তালিকা = ["লাল","সবুজ","নীল"];
ধরি কাজ_তালিকা = ["লাফাও","ঘোরাও"];

// Randomly truth নির্ধারণ
ধরি সত্য_রং = রং_তালিকা[গণিত.নিচ(গণিত.রান্ডম()*৩)];
ধরি সত্য_কাজ = কাজ_তালিকা[গণিত.নিচ(গণিত.রান্ডম()*২)];

ধরি রং = ইনপুট("সাপের রং বলো (লাল/সবুজ/নীল): ");

ফাংশন রং_গেস(){

যদি (রং == সত্য_রং){
   কাজ_গেস();
}
নাহলে{
   রং = ইনপুট("ভুল হয়েছে। আবারও সাপের রং বলো (লাল/সবুজ/নীল): ");
   রং_গেস();
}

}

ফাংশন কাজ_গেস(){

ধরি কাজ = ইনপুট("সাপটি কি করছে (লাফাও/ঘোরাও): ");

যদি (কাজ == সত্য_কাজ){

   কথা_বলো("এখন তুমি ঠিক ধরেছো!");
   কথা_বলো("সাপটি হলো " + সত্য_রং + " আর সাপটির কাজ হচ্ছে " + সত্য_কাজ);
   আঁকো("সাপ", সত্য_রং, সত্য_কাজ);
}
নাহলে{
   কাজ_গেস();
}

}

রং_গেস();
```
